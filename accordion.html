let custom =
    accordion ([],
      [ Item ([],
          (OptIcon (Some ([], Kind Forward)),
           Header "Header One",
           Body
             [ R.a [] [ R.str "Item One" ]
               R.a [] [ R.str "Item Two" ] ] ))
        Item ([],
          (OptIcon (Some ([], Kind Back)),
           Header "Header One",
           Body
             [ R.a [] [ R.str "Item One" ]
